def call(connection)
	if connection.http1?
		return UpgradeRequest.new(self, **@options).call(connection)
	elsif connection.http2?
		return ConnectRequest.new(self, **@options).call(connection)
	end
	
	raise HTTP::Error, "Unsupported HTTP version: #{connection.version}!"
end
